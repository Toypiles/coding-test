import java.util.*;
import java.io.*;
public class Main
{
	public static void main(String[] args) throws IOException {

		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());

		String a = st.nextToken();
		String b = st.nextToken();

		String s1 = reversed(a);
		String s2 = reversed(b);
        // s1,s2 길이 맞추기
		if(s1.length() > s2.length()) {
			while(s1.length() != s2.length()) {
				s2+="0";
			}
		} else {
			while(s1.length() != s2.length()) {
				s1+="0";
			}
		}

		String result = "";
		int carry =0; // 2진수 더하기 방법
		for(int i=0; i<s1.length(); i++) {
			int x = s1.charAt(i)-'0';
			int y = s2.charAt(i)-'0';
			int sum = x+y+carry;
			if(sum>=2) {
				carry =1;
			} else {
				carry=0;
			}
			if(sum%2==1) {
				result+="1";
			} else {
				result+="0";
			}
		}
		if(carry==1) {
			result+="1";
		}
		result=reversed(result); // 다시 숫자를 원 상태로 뒤집기

		int idx = 0; // 뒤집은 숫자 앞에 0이 존재할 때 없애주는 방법
		while(result.startsWith("0") && idx < result.length()-1) {
			idx++;
		}
		result = result.substring(idx);
		System.out.println(result);

	}
    // 숫자를 뒤집어서 반환
	public static String reversed(String s) {
		String result = "";
		for(int i = s.length()-1; i>=0; i--) {
			result += s.charAt(i);
		}
		return result;
	}
}
