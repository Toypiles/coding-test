import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		int N = scanner.nextInt();
		for(int i=0; i<N; i++){
		    String s = scanner.nextLine();
		    String[] sp = s.split(" ");
		    int position =2;
		    String[] sp1= new String[position];
		    String[] sp2= new String[sp.length - position];
		    for(int j=0; j<sp.length; j++){
		        if(j<position){
		            sp1[j]=sp[j];
		        }
		        else{
		            sp2[j- position] = sp[j];
		        }
		    }
		    System.out.println(sp2+" "+sp1);
		}
	}
}
